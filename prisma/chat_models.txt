
model AiChatSession {
  id        Int             @id @default(autoincrement())
  title     String
  createdAt DateTime        @default(now())
  updatedAt DateTime        @updatedAt
  messages  AiChatMessage[]
}

model AiChatMessage {
  id        Int           @id @default(autoincrement())
  sessionId Int
  role      String
  text      String
  image     String?
  createdAt DateTime      @default(now())
  session   AiChatSession @relation(fields: [sessionId], references: [id], onDelete: Cascade)
}
